# Install latest playwright
npm init playwright@latest

# Install eslint
npm install --save-dev @typescript-eslint/parser @typescript-eslint/eslint-plugin eslint typescript

# Create eslint ingnore file
touch .eslintignore

# Run eslint check (DO NOT RUN THIS INSIDE WORKSHOP REPOSITORY)
npx eslint .\
# or
npx eslint '**/*.{ts,tsx}'

# Install prettier via npm
npm install --save-dev eslint-config-prettier
npm install --save-dev --save-exact prettier

# Check code quality with prettier
npx prettier . --check

# Fix code issues with prettier
npx prettier . --write

# Install husky
npx husky-init && npm install

# Run all tests
npx playwright test 

# Start recorder
npx playwright codegen 

# In case of problems with husky install on Windows you might need to craete folder nmp in user path:
For powershell: 
mkdir $env:APPDATA\npm
For cmd:
mkdir %APPDATA%\npm
